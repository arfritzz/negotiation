%% related to attestation 
\newglossaryentry{Evidence} {
  name={Evidence},
  description={Measurements of a system to ensure
    integrity. Measurements can be used to ensure that the correct
    software/ hardware is used or that the state of the system is
    valid.}
}


\newglossaryentry{Events}
{
  name={Events},
  description= {Observable effects associated with phrase execution}
}  

\newglossaryentry{Remote Attestation}
{
  name={Remote Attestation},
  description={Consists of generating evidence of a system's integrity
	      via measurements, and reporting the evidence to a 
	      remove party for appraisal}
}

\newglossaryentry{Terms/Phrases}
{
  name={Terms/Phrases},
  description={Copland programs that are built from a small set 
	      of operators designed to orchestrate the activities
	      of measurement agents across several layers of a 
	      target system}
}

}

\newglossaryentry{Semantic Explicitness}
{
  name={Semantic Explicitness},
  description={A priniple of remote attestation that addresses the degree that measurement data contains the necessary evidence required for a trust decision}
}

\newglossaryentry{authenticity}

\newglossaryentry{correctness}

}

\newglossaryentry{protection}
{

\newglossaryentry{APB}
{
  name={APB (Attestation Protocol Block)},
  description={Attestation Protocol Blocks understand the requirements of a particular attestation situation. Can produce or evaluate measuement. Simply, it is the protocol used to collect and transmit evidence}
